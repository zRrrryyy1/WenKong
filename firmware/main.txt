// -------------------- ???? --------------------
void USART_SendChar(uint8_t ch) {
    while (USART_GetFlagStatus(USART1, USART_FLAG_TXE) == RESET);
    USART_SendData(USART1, ch);
}

void USART_SendString(char *str) {
    while (*str) USART_SendChar((uint8_t)*str++);
}

                latestTemp, set_temperature, temp_step, pwm_value);

                set_temperature, temp_step, pwm_value);
    }
    USART_SendString(buf);

// -------------------- ?????? --------------------
void ProcessReceivedData(void) {
    if (rx_index == 0) return;

    rx_buffer[rx_index] = '\0';
    uint8_t valid = 1;
    uint8_t dot_count = 0;
    for (uint8_t i = 0; i < rx_index; i++) {
        char c = rx_buffer[i];
        if (c == '-') {
            if (i != 0) { valid = 0; break; }
        } else if (c == '.') {
            dot_count++;
            if (dot_count > 1) { valid = 0; break; }
        } else if (c < '0' || c > '9') {
            valid = 0;
            break;
        }
    }

    if (valid && dot_count <= 1) {
        set_temperature = (float)atof(rx_buffer);
        USART_SendString(" Set ok\r\n");
    } else {
        USART_SendString(" Invalid\r\n");
    }

    rx_index = 0;
    rx_has_data = 0;
    timeout_counter = 0;
}

// -------------------- USART1 ?? --------------------
void USART1_IRQHandler(void) {
    if (USART_GetITStatus(USART1, USART_IT_RXNE) != RESET) {
        uint8_t rec = USART_ReceiveData(USART1);
        if ((rec >= '0' && rec <= '9') || rec == '-' || rec == '.') {
            if (rx_index < RX_BUF_SIZE - 1) {
                rx_buffer[rx_index++] = rec;
            } else {
                rx_index = 0;
            }
        }
        rx_has_data = 1;
        timeout_counter = 0;
    }
}

}

// -------------------- ????(SysTick???,10ms) --------------------
void KEY_Scan(void) {
    static struct {
        uint8_t state;
        uint8_t cnt;
        uint8_t short_flag;
    } key[3] = {0};

    uint8_t cur[3];
    cur[0] = (GPIO_ReadInputDataBit(KEY_PORT, KEY_STEP_PIN) == Bit_RESET) ? 1 : 0;
    cur[1] = (GPIO_ReadInputDataBit(KEY_PORT, KEY_UP_PIN)   == Bit_RESET) ? 1 : 0;
    cur[2] = (GPIO_ReadInputDataBit(KEY_PORT, KEY_DOWN_PIN) == Bit_RESET) ? 1 : 0;

    for (int i = 0; i < 3; i++) {
        switch (key[i].state) {
            case 0: if (cur[i]) { key[i].state = 1; key[i].cnt = 1; } break;
            case 1:
                if (cur[i]) {
                    if (++key[i].cnt >= DEBOUNCE_CNT) {
                        key[i].state = 2;
                        key[i].short_flag = 1;
                    }
                } else key[i].state = 0;
                break;
            case 2: if (!cur[i]) key[i].state = 0; break;
        }
    }

    if (key[0].short_flag) {            // ????
        key[0].short_flag = 0;
        step_index = (step_index + 1) % STEP_COUNT;
        temp_step = step_table[step_index];
    }
    if (key[1].short_flag) {            // ??
        key[1].short_flag = 0;
        set_temperature += temp_step;
    }
    if (key[2].short_flag) {            // ??
        key[2].short_flag = 0;
        set_temperature -= temp_step;
    }
}

// -------------------- TIM2 PWM ??? (20kHz, PA1) --------------------
void TIM2_PWM_Init(void) {
    TIM_TimeBaseInitTypeDef  TIM_TimeBaseStructure;
    TIM_OCInitTypeDef        TIM_OCInitStructure;

    TIM_TimeBaseStructure.TIM_Period = PWM_PERIOD;
    TIM_TimeBaseStructure.TIM_Prescaler = PWM_PRESCALER;
    TIM_TimeBaseStructure.TIM_ClockDivision = 0;
    TIM_TimeBaseStructure.TIM_CounterMode = TIM_CounterMode_Up;
    TIM_TimeBaseInit(HEAT_TIM, &TIM_TimeBaseStructure);

    TIM_OCInitStructure.TIM_OCMode = TIM_OCMode_PWM1;
    TIM_OCInitStructure.TIM_OutputState = TIM_OutputState_Enable;
    TIM_OCInitStructure.TIM_Pulse = DUTY_STOP;   // ??0
    TIM_OCInitStructure.TIM_OCPolarity = TIM_OCPolarity_High;
    TIM_OC2Init(HEAT_TIM, &TIM_OCInitStructure);

    TIM_OC2PreloadConfig(HEAT_TIM, TIM_OCPreload_Enable);
    TIM_ARRPreloadConfig(HEAT_TIM, ENABLE);
    TIM_Cmd(HEAT_TIM, ENABLE);
}

// -------------------- ???? (PWM) --------------------
void HeatingControl(void) {
    if (latestTemp <= -900.0f) {
        heater_duty = DUTY_STOP;
    } else if (latestTemp < set_temperature) {
        heater_duty = DUTY_HEAT;  // 50% = 24
    } else {
        heater_duty = DUTY_STOP;
    }
    TIM_SetCompare2(HEAT_TIM, heater_duty);
    pwm_value = heater_duty;   // ??????
}

    SysTick_CLKSourceConfig(SysTick_CLKSource_HCLK_Div8);
    if (SysTick_Config(SystemCoreClock / 8 / 100)) {
        while (1);
    }
}

// -------------------- SysTick ?? (10ms) --------------------
void SysTick_Handler(void) {
    static uint16_t cnt_500ms = 0;
    static uint16_t cnt_1s = 0;

    KEY_Scan();

    // ???? (200ms)
    if (rx_has_data && rx_index > 0) {
        timeout_counter++;
        if (timeout_counter >= 20) {
            ProcessReceivedData();
            rx_has_data = 0;
            timeout_counter = 0;
        }
    }

    // ??????? (500ms)
    cnt_500ms++;
    if (cnt_500ms >= 50) {
        cnt_500ms = 0;
        flag_temp_update = 1;
    }

    // ??????? (1s)
    cnt_1s++;
    if (cnt_1s >= 900) {
        cnt_1s = 0;
        flag_1s = 1;
    }
}

}
